182  for(auto& [id, v, info] : _modes) {
183  if(nid == id) {
184  nv = nv + 1;
185  }
186  }
187  _modes.push_back({nid, nv, {}});
188  return &std::get<2>(_modes.back());
189  }
190  return nullptr;
191  }
